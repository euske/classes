小課題 6. 投票集計プログラム vote3.py - 解答例

vote3.py 
# 候補者リスト。
cands = ["abe", "trump", "kim"]
# 投票数リスト。
votes = [0, 0, 0]

while True:
    # 候補者の名前を入力する。空ならばループを抜ける。
    name = input("name:")
    if name == "":
        break
    # name を各候補者の名前と比較し、見つかった番号を found に入れる。
    found = -1
    for i in range(len(cands)):
        if name == cands[i]:
            found = i
            break
    if 0 <= found:
        # その候補者の票を 1だけ増やす。
        votes[found] = votes[found] + 1
    else:
        # 見つからない場合、found は -1 のままなので、エラーを表示。
        print("invalid name")

# 3人の得票数を表示する。
for i in range(3):
    print(f"{cands[i]}: {votes[i]}")
