小課題 3. 引き算および割り算をするプログラム - 解答例

sub.py 
a = int(input("a:"))
b = int(input("b:"))
while 0 < b:
    a = a - 1
    b = b - 1
print(f"answer={a}")
div.py 
x = int(input("x:"))
y = int(input("y:"))
q = 0
while y <= x:
    b = y
    while 0 < b:
        x = x - 1
        b = b - 1
    q = q + 1
print(f"quotient = {q}")
print(f"remainder = {x}")
